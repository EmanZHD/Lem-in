package main

import "fmt"
func processGroups(groups [][][]string) {
    for i, group := range groups {
        fmt.Printf("Group %d:\n", i+1)
        for _, subGroup := range group {
            fmt.Println("  Subgroup:", subGroup)
        }
    }
}

func main() {
    groups := [][][]string{
        {
            {"h", "A", "c", "k", "end"},
            {"t", "E", "a", "m", "end"},
            {"0", "o", "n", "e", "end"},
        },
        {
            {"h", "n", "m", "end"},
        },
        {
            {"h", "n", "e", "end"},
            {"t", "E", "a", "m", "end"},
        },
    }
    processGroups(groups)
}
